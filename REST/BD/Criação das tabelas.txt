-- Criação da tabela produtos
CREATE TABLE produtos (
cd_produto INTEGER,
nm_produto VARCHAR(50),
ds_produtoo VARCHAR(70),
vl_venda DECIMAL(8,2),
vl_custo DECIMAL(8,2),
qt_estoque INTEGER,
PRIMARY KEY (cd_produto)
);


-- Criação da tabela lojas
CREATE TABLE lojas (
cd_loja INTEGER,
nm_loja VARCHAR(50),
ds_loja VARCHAR(70),
cd_produto INTEGER,
PRIMARY KEY (cd_loja),
FOREIGN KEY (cd_produto) REFERENCES produtos (cd_produto)
);
